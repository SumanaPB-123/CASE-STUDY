#Creating User Table

3)CREATE TABLE user (USER_ID int, 
                     PASSWORD varchar(10),
                     USERNAME varchar(15),
                     EMAIL varchar(30),
                     FIRST_NAME varchar(15),
                     LAST_NAME varchar(15),
                     CONTACT_NUMBER bigint,
                     ROLE varchar(10),
                     IS_ACTIVE boolean,
                     DOB date,
                     CREATED_ON datetime,
                     PRIMARY KEY (USER_ID))

#Creating Task Table

5)CREATE TABLE task (TASK_ID int,
                     OWNER_ID int,
                     CREATOR_ID int,
                     NAME varchar(30),
                     DESCRIPTION varchar(30),
                     STATUS varchar(30),
                     PRIORITY varchar(30),
                     NOTES varchar(30),
                     ISBOOKMARKED boolean,
                     CREATED_ON datetime,
                     STATUS_CHANGED_ON datetime,
                     PRIMARY KEY (TASK_ID),
                     FOREIGN KEY (OWNER_ID) references USER(USER_ID),
                     FOREIGN KEY (CREATOR_ID) references USER(USER_ID) on DELETE CASCADE)
